

import React from 'react';
import { Sailboat, Zap, Palette, ChevronDown, Lock } from 'lucide-react';

function ModuleIcon({ type }: { type: string }) {
  switch (type) {
    case 'basics':
      return <Sailboat className="text-blue-500" />;
    case 'core':
      return <Zap className="text-yellow-500" />;
    case 'advanced':
      return <Palette className="text-green-500" />;
    default:
      return null;
  }
}

function LessonBadge({ type }: { type: string }) {
  const badgeClass = `px-2 py-1 text-xs font-semibold rounded-full`;
  switch (type) {
    case 'FREE PREVIEW':
      return <span className={`${badgeClass} bg-green-100 text-green-800`}>{type}</span>;
    case 'DIVE STRAIGHT IN THE DEEP END':
      return <span className={`${badgeClass} bg-yellow-100 text-yellow-800`}>{type}</span>;
    case 'PROMPTS, PARAMETERS, SCRIPTS & MORE':
      return <span className={`${badgeClass} bg-purple-100 text-purple-800`}>{type}</span>;
    default:
      return null;
  }
}

function SubLessonItem({ title, locked }: { title: string; locked: boolean }) {
  return (
    <div className="flex items-center justify-between py-2">
      <span className="text-sm">{title}</span>
      {locked && <Lock className="w-4 h-4 text-gray-400" />}
    </div>
  );
}

function Module({ number, title, type, lessons, icon }: { number: number; title: string; type: string; lessons: number; icon: string }) {
  return (
    <div className="bg-white rounded-lg shadow-md p-4 mb-4">
      <div className="flex items-center mb-2">
        <div className="w-10 h-10 rounded-full bg-gray-100 flex items-center justify-center mr-3">
          <ModuleIcon type={icon} />
        </div>
        <div>
          <h3 className="text-sm text-gray-500">MODULE {number}</h3>
          <h2 className="text-lg font-semibold">{title}</h2>
        </div>
      </div>
      <LessonBadge type={type} />
      <div className="flex items-center justify-between mt-4">
        <span className="text-sm text-gray-600">{lessons} LESSONS</span>
        <ChevronDown className="w-5 h-5 text-gray-400" />
      </div>
    </div>
  );
}

export default function Page() {
  return (
    <div className="max-w-2xl mx-auto p-4">
      <Module 
        number={0}
        title="Essential Prerequisites"
        type="FREE PREVIEW"
        lessons={2}
        icon="basics"
      />
      <Module 
        number={1}
        title="Proprietary Models & Diffusion"
        type="DIVE STRAIGHT IN THE DEEP END"
        lessons={4}
        icon="core"
      />
      <Module 
        number={2}
        title="Intro to Stable Diffusion"
        type="PROMPTS, PARAMETERS, SCRIPTS & MORE"
        lessons={0}
        icon="advanced"
      />
      {/* You can add SubLessonItems here if needed */}
    </div>
  );
}
